402 template<typename T1>
403 Quaternion operator/=(const Quaternion<T1>& y) {
404
405 T n2 = y.norm_squared();
406
407 T at = _a * y.a() + _b * y.b() + _c * y.c() + _d * y.d();
408 T bt = -_a * y.b() + _b * y.a() - _c * y.d() + _d * y.c();

410 T dt = -_a * y.d() - _b * y.c() + _c * y.b() + _d * y.a();
411
412 _a = at / n2;
413 _b = bt / n2;
414 _c = ct / n2;
415 _d = dt / n2;

418 }

454template<typename T>
455inline Quaternion<T> spherical(T rho, T theta, T phi1, T phi2) {
456
457 T d = std::sin(phi2);
458 T cr = std::cos(phi2);
459 T c = cr * std::sin(phi1);
460 cr *= std::cos(phi1);
461 T b = cr * std::sin(theta);
462 T a = cr * std::cos(theta);
463
464 return {rho * a, rho * b, rho * c, rho * d};
465}

524template<typename T>
525inline polar_representation<T> to_polar_representation(const Quaternion<T>& x) {
526 T nu = x.unreal_norm_squared();
527 T n = std::sqrt(nu + x.a() * x.a());
528 assert(nu >= 0);
529 if (nu > 0) {
530 T theta = std::acos(x.a() / n);
531 T ns = sqrt(nu);
532 return {{n, theta, x.b() / ns, x.c() / ns, x.d() / ns}};
533 }
534 const T pi = std::atan2(+0., -0.);
535 // theta = 0 or pi, because n = +/- a().
536 return {{n, n == x.a() ? 0 : pi, 0, 0, 0}};
537}

616template<typename T>
617inline rotation_matrix<T> to_rotation_matrix(const Quaternion<T>& x) {
618 // 21 operations?
619 T a2 = x.a() * x.a(), b2 = x.b() * x.b(), c2 = x.c() * x.c(), d2 = x.d() * x.d();
620 T ab = x.a() * x.b(), ac = x.a() * x.c(), ad = x.a() * x.d();
621 T bc = x.b() * x.c(), bd = x.b() * x.d();
622 T cd = x.c() * x.d();
623 std::array<T, 3> r0{{a2 + b2 - c2 - d2, 2 * (bc - ad), 2 * (bd + ac)}};
624 std::array<T, 3> r1{{2 * (bc + ad), a2 - b2 + c2 - d2, 2 * (cd - ab)}};
625 std::array<T, 3> r2{{2 * (bd - ac), 2 * (cd + ab), a2 - b2 - c2 + d2}};
626 return {{r0, r1, r2}};
627}

629template<typename T>
630inline Quaternion<T> from_rotation_matrix(const rotation_matrix<T>& rm) {
631 T t = rm[0][0] + rm[1][1] + rm[2][2];
632 if (t > 0) {
633 T s = 0.5 / std::sqrt(t + 1);
634 return {0.25 / s,
635 (rm[2][1] - rm[1][2]) * s,
636 (rm[0][2] - rm[2][0]) * s,
637 (rm[1][0] - rm[0][1]) * s};
638 } else {
639 if (rm[0][0] > rm[1][1] && rm[0][0] > rm[2][2]) {
640 T s = 2.0 * std::sqrt(1.0 + rm[0][0] - rm[1][1] - rm[2][2]);
641 return {(rm[2][1] - rm[1][2]) / s,
642 0.25 * s,
643 (rm[0][1] + rm[1][0]) / s,
644 (rm[0][2] + rm[2][0]) / s};
645 } else if (rm[1][1] > rm[2][2]) {
646 T s = 2.0 * std::sqrt(1.0 + rm[1][1] - rm[0][0] - rm[2][2]);
647 return {(rm[0][2] - rm[2][0]) / s,
648 (rm[0][1] + rm[1][0]) / s,
649 0.25 * s,
650 (rm[1][2] + rm[2][1]) / s};
651 } else {
652 T s = 2.0 * std::sqrt(1.0 + rm[2][2] - rm[0][0] - rm[1][1]);
653 return {(rm[1][0] - rm[0][1]) / s,
654 (rm[0][2] + rm[2][0]) / s,
655 (rm[1][2] + rm[2][1]) / s,
656 0.25 * s};
657 }
658 }

667template <typename T>
668inline std::array<T, 3> to_euler(const Quaternion<T>& q) {
669 // roll (x-axis rotation)
670 T sinr_cosp = 2 * (q.a() * q.b() + q.c() * q.d());
671 T cosr_cosp = 1 - 2 * (q.b() * q.b() + q.c() * q.c());
672 T roll = std::atan2(sinr_cosp, cosr_cosp);
673
674 // pitch (y-axis rotation)
675 T sinp = std::sqrt(1 + 2 * (q.a() * q.c() - q.b() * q.d()));
676 T cosp = std::sqrt(1 - 2 * (q.a() * q.c() - q.b() * q.d()));
677 T pitch = 2 * std::atan2(sinp, cosp) - M_PI / 2;
678
679 // yaw (z-axis rotation)
680 T siny_cosp = 2 * (q.a() * q.d() + q.b() * q.c());
681 T cosy_cosp = 1 - 2 * (q.c() * q.c() + q.d() * q.d());
682 T yaw = std::atan2(siny_cosp, cosy_cosp);
683 return {roll, pitch, yaw};
684}

691template <typename T>
692inline Quaternion<T> from_euler(const std::array<T, 3>& x) {
693 T cr = std::cos(x[0] * 0.5);
694 T sr = std::sin(x[0] * 0.5);
695 T cp = std::cos(x[1] * 0.5);
696 T sp = std::sin(x[1] * 0.5);
697 T cy = std::cos(x[2] * 0.5);
698 T sy = std::sin(x[2] * 0.5);
699 return {
700 cr * cp * cy + sr * sp * sy,
701 sr * cp * cy - cr * sp * sy,
702 cr * sp * cy + sr * cp * sy,
703 cr * cp * sy - sr * sp * cy
704 };
705}

717template <typename T>
718struct hash : public std::unary_function<Quaternion<T>, size_t> {
719
720 inline size_t operator()(const Quaternion<T>& x) const {
721
722 auto mix = [](uint64_t h) {
723 (h) ^= (h) >> 23;
724 (h) *= 0x2127599bf4325c37ULL;
725 (h) ^= (h) >> 47;
726 return h;
727 };
728
729 const uint64_t len = 4 * sizeof(T); // in bytes, Qf is 4*4, Qd is 4*8, and Qld 4*16
730 const uint64_t m = 0x880355f21e6d1965ULL;
731 const uint64_t *pos = (const uint64_t *) &x;
732 const uint64_t *end = pos + (len / 8);
733 uint64_t h = 31 ^(len * m);
734 uint64_t v;
735
736 while (pos != end) {
737 v = *pos++;
738 h ^= mix(v);
739 h *= m;
740 }
741
742 return mix(h);
743 }
744};

1143 if (x.a() > 0)

1220template<typename T>
1221inline Quaternion<T> pow(const Quaternion<T>& x, int expt) {
1222
1223 if (expt < 0)
1224 return inverse(pow(x, -expt));
1225 if (expt == 0)
1226 return {1};
1227 if (expt == 1)
1228 return x;
1229 if (expt == 2)
1230 return pow2(x);
1231 if (expt == 3)
1232 return pow3(x);
1233 if (expt == 4)
1234 return pow4(x);
1235
1236 Quaternion<T> x4 = pow4(x), y = x4;
1237 for (int i = 1; i < expt / 4; ++i)
1238 y *= x4;
1239 if (expt % 4 == 3)
1240 y *= pow3(x);
1241 if (expt % 4 == 2)
1242 y *= pow2(x);
1243 if (expt % 4 == 1)
1244 y *= x;
1245 return y;
